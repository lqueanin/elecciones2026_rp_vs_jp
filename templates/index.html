async function cargarDatos() { const ONPE_URL = 'https://resultadoelectoral.onpe.gob.pe/presentacion-backend/resumen-general/participantes?idEleccion=10&tipoFiltro=eleccion'; const PROXY = 'https://api.allorigins.win/get?url='; try { // 1. El navegador (tu IP limpia) intenta bajar los datos de la ONPE const resProxy = await fetch(PROXY + encodeURIComponent(ONPE_URL)); const jsonProxy = await resProxy.json(); const onpeData = JSON.parse(jsonProxy.contents); if (onpeData.success) { const rp = onpeData.data.find(p => p.codigoAgrupacionPolitica == 35); const jpp = onpeData.data.find(p => p.codigoAgrupacionPolitica == 10); // 2. "Donamos" el dato al servidor de Railway para guardarlo en la DB await fetch('/api/push-votos?token=unjbg_esis_2026', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ votos_rp: rp.totalVotosValidos, votos_jpp: jpp.totalVotosValidos }) }); } } catch (e) { console.log("No se pudo actualizar desde el cliente, cargando historial..."); } // 3. Finalmente, cargamos el historial de la base de datos para mostrar la gráfica const resDB = await fetch('/api/historial'); const historial = await resDB.json(); if (historial.length > 0) { const ultimo = historial[historial.length - 1]; document.getElementById('votos-rp').innerText = ultimo.votos_rp.toLocaleString('es-PE'); document.getElementById('votos-jpp').innerText = ultimo.votos_jpp.toLocaleString('es-PE'); document.getElementById('diferencia-votos').innerText = ultimo.diferencia.toLocaleString('es-PE') + " votos"; const etiquetas = historial.map(h => h.fecha); const datos = historial.map(h => h.diferencia); renderizarGrafico(etiquetas, datos); } }
